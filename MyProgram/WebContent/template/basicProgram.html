MethodOverloading Example 

public class MethodOverloading_AddNumber
{
    public static void main(String[] args) 
    {
        System.out.println("Execution start");
        sum(2,4);
        sum(55);
        sum(2,5,9);

    }
    static void sum(int a)
    {
        System.out.println("execution sum "+a);
    }
    static void sum(int a, int b)
    {
        System.out.println("execution sum(int a ,int b) "+(a+b));
    }
    static void sum(int a, int b,int c)
    {
        System.out.println("execution sum(int a ,int b,int c) "+(a+b+c));
    } 
}

Swap Two number 

public class SwapingNumber {
    public static void main(String [] args){
        swap(5,6);
    }
    public static void swap(int a, int b){
        int temp =a;
        a=b;
        b=temp;
        System.out.println("a= "+a+", b="+b);
    }
}
